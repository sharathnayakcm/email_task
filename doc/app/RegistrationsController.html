def update
  @user = User.find(current_user.id)
  if @user.update_attributes(params[:user])
    sign_in @user, :bypass => true
    flash[:notice] = "Profile has been saved successfully"
    redirect_to dashboard_home_index_path
  else
    render "edit"
  end
end
