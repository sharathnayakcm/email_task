def self.find_for_database_authentication(warden_conditions)
  conditions = warden_conditions.dup
  email = self.valid_email(conditions.delete(:email))
  where(conditions).where(["lower(email) = :value and is_active = true", { :value => email.strip.downcase }]).first
end

def buzz_count(channels)
  channel_list = []
  channels.each do |c|
    buzz_count = get_unsync_count(c)
    channel_list << {:id => c.id, :name => c.name, :buzz_count => buzz_count, :is_cug => c.is_cug}
  end
  # Sorting channel list according to count of unread buzzes
  channel_list.sort{|a,b| [b[:buzz_count], a[:name].downcase]  <=> [a[:buzz_count], b[:name].downcase]}
end

def cug_insync_stats_count(is_core,summary)
  unsynced_buzzes = 0
  unsynced_cugs = 0
  cugs = Channel.includes(:subscriptions => :user).where("is_cug = ? and users.id = ? and channels.is_active = ? and is_core = ?", true, self.id, true, is_core)
  buzz_count(cugs).each do |c|
    if c[:buzz_count] > 0
      unsynced_buzzes += c[:buzz_count]
      unsynced_cugs += 1
    end
  end
   #"#{unsynced_cugs} #{is_core ? 'core' : 'peripheral'} CUGs (#{unsynced_buzzes} buzzes)"
   summary == true ? unsynced_buzzes : unsynced_cugs
end

def cug_list(is_core, limit = 9, hide_sync = false, all = false)
  cugs = Channel.includes(:subscriptions => :user).where("is_cug = ? and users.id = ? and channels.is_active = ? and is_core = ?", true, self.id, true, is_core)
  cugs = hide_sync ? buzz_count(cugs.uniq).select{|c| c[:buzz_count] > 0} : buzz_count(cugs.uniq)
  cugs = (all || limit == 0) ? cugs : cugs.take(limit)
end

def get_unsync_count channel
   if channel.is_cug
      ## need to refactor the logic
      last_insyced_id = channel.get_channel_last_insynced_buzz_id(self)
      unsync_buzzs = Buzz.where("channel_id = ? and id > ? and user_id != ?",channel.id, last_insyced_id, self.id).collect{|b| b.id}
      buzz_com = BuzzMember.where("channel_id = ? and buzz_id > ? and user_id !=?", channel.id,last_insyced_id,self.id).collect{|b| b.buzz_id}
      buzz_mem = BuzzMember.where("channel_id = ? and buzz_id > ? and user_id =?", channel.id,last_insyced_id,self.id).collect{|b| b.buzz_id}
      bid = []
      buzz = buzz_com - buzz_mem
      bid << buzz[0]
      bz = buzz[0]
      buzz.each do |b|
        if bz != b
         bid << b
         bz = b
        end
      end
      buzz_count= (unsync_buzzs - bid).size
    elsif channel.subscriptions.map(&:user_id).include?(self.id)
      last_viewed = channel.get_channel_last_viewed(self)
      buzz_count = (last_viewed.blank? || last_viewed[0].nil?) ? Buzz.where("channel_id = ? and user_id != ?", channel.id, self.id).count : Buzz.where("channel_id = ? and created_at > ? and user_id != ?", channel.id, last_viewed, self.id).count
    else
      buzz_count = Buzz.where("channel_id = ? and created_at > ? and user_id != ?", channel.id, APP_CONFIG['buzz']['new_buzzes'].hours.ago, self.id).count
    end
   buzz_count
end
