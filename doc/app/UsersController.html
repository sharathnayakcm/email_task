def save_csv
  err_msg = []
  csv_file = params[:upload_csv][:csv].read
  begin
    CSV.parse(csv_file, :headers => :first_row, :return_headers => false) do |row|
      email = User.valid_email(row[2])
      if User.is_valid_email?(email)
        user = User.new(:first_name => row[0], :last_name => row[1], :email => email)
        user.new_random_password
        if !user.save
          err_msg << "$#{user.errors.full_messages[0]} for #{row[2]}"
        end
      else
        err_msg << "$Invalid email #{row[2]}"
      end
    end
    flash[:notice] = err_msg.empty? ? "New users invited successfully" : err_msg
  rescue CSV::MalformedCSVError
    flash[:notice] = "$Malformed CSV! Please check syntax"
  end
  
  redirect_to users_path
end
