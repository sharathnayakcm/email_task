def add_to_watch
  @update_div = params[:update_div]
  @channel = Channel.find_by_id(params[:channel_id])

      @message = WatchChannel.add_to_watch(@channel,current_user)

    else
      @message = WatchChannel.remove_from_watch(@channel,current_user)
      @action_status = @channel.channel_in_watch?(current_user)
    end
  else
    @message = @channel.errors.full_messages
  end
end

def calendar_buzzes
  @is_cug = (params[:is_cug] == 'true')
  @error = ""
  @buzzes = []
  @div_update_id = params[:update_div_id]
  start_date = params[:start_date]
  end_date = params[:end_date]
  date_diff = (Date.parse(end_date) - Date.parse(start_date)).to_i
  if Date.parse(start_date) <= Date.today && Date.parse(end_date) <= Date.today
    if Date.parse(start_date) <= Date.parse(end_date)
      if date_diff <= APP_CONFIG['date']['max_calendar_days_diff'].to_i
        @buzzes = current_user.calendar_buzzes(@is_cug, start_date, end_date)
        @error = "No buzzes found for selected dates in #{@is_cug ? 'CUGs' : 'Channels'}" unless @buzzes.count > 0
      else
        @error = "$Date difference should not exceed #{APP_CONFIG['date']['max_calendar_days_diff']} days"
      end
    else
      @error = "$Please enter valid date range"
    end
  else
    @error = "$Date should not be greater than today's date"
  end
end

def process_command
  if !params[:beehive][:command].blank?
    command = params[:beehive][:command]
    attachment = params[:beehive][:attachfile]
    if chk_format(command)
      if is_here_command?(command)
        channel = nil
        channel = Channel.find_by_id params[:beehive][:channel_id].to_i unless params[:beehive][:channel_id].blank?
        process_result = process_here_command(command.clone, attachment, channel)
      else
        process_result = process_command_task(command.clone, attachment)
      end
      if process_result[:task] == 'invalid_here'
        flash[:notice] = process_result[:process_msg]
        redirect_to request.env['HTTP_REFERER']
        flash[:command] = command
      elsif process_result[:show_data] && process_result[:task] == 'find'
        if process_result[:apply_to] == 'insync'
          redirect_to :action => 'insync', :apply_to => process_result[:apply_to], :process_values => process_result[:process_values]
        else
          redirect_to :action => 'search', :apply_to => process_result[:apply_to], :process_values => process_result[:process_values]
        end
      elsif process_result[:task] == 'watch'
        flash[:notice] = process_result[:process_msg][:msg]
        flash[:command] = process_result[:process_msg][:success_status] ? "" : command
        redirect_to home_index_path(:command => command)
      elsif process_result[:task] == 'subscription' || process_result[:task] == 'buzzout'
        flash[:notice] = process_result[:process_msg][:msg]
        flash[:command] = process_result[:process_msg][:success_status] ? "" : command
        channel_id =  process_result[:process_msg][:channel_id]
        redirect_to channel_id == 0 ? request.env['HTTP_REFERER'] : buzzes_home_path(channel_id)
      elsif process_result[:task] == 'add_channel' || process_result[:task] == 'remove_channel' || process_result[:task] == 'unsubscription'
        if process_result[:apply_to] == 'channel'
          flash[:notice] = process_result[:process_msg][:msg]
          flash[:command] = process_result[:process_msg][:success_status] ? "" : command
          redirect_to channels_home_index_path
        elsif process_result[:apply_to] == 'cug'
          flash[:notice] = process_result[:process_msg][:msg]
          flash[:command] = process_result[:process_msg][:success_status] ? "" : command
          redirect_to cugs_home_index_path
        else
          flash[:command] = process_result[:process_msg][:success_status] ? "" : command
          redirect_to dashboard_home_index_path(:notice => process_result[:process_msg][:msg], :command => flash[:command])
        end
      elsif process_result[:task] == 'tag' || process_result[:task] == 'merge'
        flash[:notice] = process_result[:process_msg][:msg]
        flash[:command] = process_result[:process_msg][:success_status] ? "" : command
        redirect_to request.env['HTTP_REFERER']
      else
        redirect_to dashboard_home_index_path(:notice => process_result[:process_msg])
      end
    else
      flash[:notice] = ["$Invalid command"]
      flash[:command] = command
      redirect_to request.env['HTTP_REFERER']
    end
  else
    flash[:notice] = ["$Invalid command"]
    redirect_to request.env['HTTP_REFERER']
  end
end

def search
  @search_count = 0
  search_data = process_find(params[:apply_to], params[:process_values])
  @search_keyword = params[:process_values]
  @search_type = search_data[:search_type]
  @search_data = search_data[:search_data]
  if params[:apply_to] == 'buzz'
    @search_data.map{|c|@search_count += c[:buzz_count]}
  else
    @search_count = @search_data.count
  end
  flash[:notice] = "#{@search_count} #{@search_count == 1 ? 'record' : 'records'} found"
  render 'search_buzzes' if params[:apply_to] == 'buzz'
end

def sub_unsub_channel
  @watch_div = params[:watch_div]
  @update_div = params[:update_div]
  @action = params[:act]
  @channel = Channel.find_by_id(params[:channel_id])
  @action_status = false
  if !@channel.nil?
    if @action == 'subscribe'
      @message, @action_status = Subscription.subscribe_me(@channel, current_user)
    else
      @message, @action_status = Subscription.unsubscribe_me(@channel, current_user)
    end
  else
    @message = @channel.errors.full_messages
  end
end
