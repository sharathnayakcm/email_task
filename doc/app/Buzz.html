def self.associated_channel_buzz_post(buzz)
  associated_channels = ChannelAssociation.where(:cug_id => buzz.channel_id)
  associated_channels.each do |acug|
    associated_buzz = Buzz.new(:message => buzz.message, :channel_id => acug.channel_id, :user_id => buzz.user_id, :priority => buzz.priority, :attachment => buzz.attachment)
    associated_buzz.save
  end
end

def self.channel_buzzout(channel_name, is_cug, buzz_msg, priority, user, apply_to, attachment = nil)
  msg = []
  success_status = false
  channel_id = 0
  if apply_to == 'alias'
    channel = Channel.joins(:channel_aliases).where("channel_aliases.name = ? and channels.is_active = ?", channel_name, true).first
  else
    channel = Channel.where("name = ? and is_cug = ? and is_active = ?", channel_name, is_cug, true).first
  end
  if !channel.nil?
    channel_id = channel.id
    if channel.is_admin?(user) || channel.is_subscribed?(user)
      buzz = Buzz.new(:message => buzz_msg, :channel_id => channel.id, :user_id => user.id, :priority => priority, :attachment => attachment)
      if buzz.valid?
        buzz.save
        associated_channel_buzz_post(buzz)
        msg << "Buzzed out to #{channel.channel_or_cug} #{channel.name} successfully#{' marked as priority buzz' if priority}"
        success_status = true
      else
        buzz.errors.full_messages.each do |e|
          msg << "$#{e}" unless msg.include?(e)
        end
      end
    else
      msg << "$You must be subscribed to or owner of the #{channel.channel_or_cug} #{channel.name}"
    end
  else
    msg << "$#{is_cug ? 'CUG' : 'Channel'} doesn't exist or inactive"
  end
  return msg, channel_id, success_status
end

def self.channel_rezzout(channel_name, is_cug, buzz_msg, priority, user, apply_to, attachment = nil, buzz_id)
  msg = []
  success_status = false
  channel_id = 0
  channel = Channel.get_channel(channel_name, apply_to, is_cug)
  if !channel.nil?
    channel_id = channel.id
    if channel.is_admin?(user) || channel.is_subscribed?(user)
      buzz = channel.buzzs.find(buzz_id)
      if buzz
        buzz = buzz.children.new(:message => buzz_msg, :channel_id => channel.id, :user_id => user.id, :priority => priority, :attachment => attachment)
        if buzz.valid?
          buzz.save
          msg << "Buzzed out to #{channel.channel_or_cug} #{channel.name} successfully#{' marked as priority buzz' if priority}"
          success_status = true
        else
          buzz.errors.full_messages.each do |e|
            msg << "$#{e}" unless msg.include?(e)
          end
        end
      else
        msg << "Please enter a valid buzz id"
      end
    else
      msg << "$You must be subscribed to or owner of the #{channel.channel_or_cug} #{channel.name}"
    end
  else
    msg << "$#{is_cug ? 'CUG' : 'Channel'} doesn't exist or inactive"
  end
  return msg, channel_id, success_status
end

def self.get_all_collapsed_buzz(buzzs)
  collapsed_buzzs = []
  for buzz in buzzs
    collapsed_buzzs << buzz unless buzz.ancestry
  end
  return Channel.buzz_data(collapsed_buzzs)
end

 def self.get_all_expanded_buzzs(buzzs,search_text)
   expanded_buzzs = []
   for buzz in buzzs
     parent_buzz = buzz.ancestry ? buzz.parent : buzz
     children_buzz = parent_buzz.children.where("message like ?", "%#{search_text}%")
     expanded_buzz << {:parent => parent_buzz, :children => children_buzz}
   end
   return expanded_buzzs
end

def self.remove_buzz(buzz_id)
  msg = []
  buzz = Buzz.find(buzz_id)
  if !buzz.nil?
    buzz.destroy
    msg << 'This buzz has been deleted successfully.'
  else
    msg << buzz.errors.full_messages
  end
  msg
end

def insynced?(user)
  if self.channel.is_cug
    buzz_insync = BuzzInsync.where(:channel_id => self.channel_id, :user_id => user.id).first
    insync_buzz_id = buzz_insync.nil? ? 0 : buzz_insync.buzz_id
    self.user_id != user.id ? (self.id <= insync_buzz_id) : true
  end
end
