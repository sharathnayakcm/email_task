def create
  channel = @buzz.channel.is_cug ? "@#{@buzz.channel.name}" : "##{@buzz.channel.name}"
  command = "r#{channel} #{@buzz.id} #{params[:rezz][:message]}"
  @process_result = process_command_task(command, params[:rezz][:attachfile])
  @rezzs = @buzz.children
end
