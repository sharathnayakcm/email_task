def self.subscribe_me(channel, user)
  msg = []
  success_status = false
  return "$You can not subscribe your own #{channel.channel_or_cug}" if  channel.user_id == user.id
  subscription = Subscription.new(:channel_id => channel.id, :user_id => user.id, :is_core => channel.is_cug ? true : false)
  if subscription.valid?
    subscription.save
  else
    subscription.errors.each do |attr, val|
      msg << "$#{val}" unless msg.include?(val)
    end
  end
  msg << "You have been subscribed successfully to #{channel.channel_or_cug}  #{channel.name}" unless !msg.blank?
  success_status = true
  return msg, success_status
end

def self.subscribe_users(channel, users)
  msg = []
  success_status = false
  users_name = ""
  invalid_users = ""
  users.each do |email|
    email = User.valid_email(email)
    user = User.find_by_email email
    if !user.nil?
      if channel.user_id != user.id
        if Subscription.where(:channel_id => channel.id, :user_id => user.id).count > 0
          msg << "#{user.full_name} is already subscribed to this #{channel.channel_or_cug}"
        else
          subscription = Subscription.new(:channel_id => channel.id, :user_id => user.id, :is_core => channel.is_cug ? true : false)
          if subscription.valid?
            subscription.save
            users_name += user.full_name + ", "
          else
            subscription.errors.full_messages.each do |e|
              msg << "$#{e}" unless msg.include?(e)
            end
          end
        end
      else
        msg << "$You can not subscribe your own #{channel.channel_or_cug}"
      end
    else
      invalid_users += email + ", "
      #msg << "User not found with the email id #{email}"
    end
  end
  msg << "#{users_name[0, users_name.length - 2]} added successfully to the #{channel.channel_or_cug} #{channel.name}" if !users_name.blank?
  if !invalid_users.blank?
    msg << "$#{invalid_users[0, invalid_users.length - 2]} not found"
  else
    msg << "You have been subscribed successfully" unless !msg.blank?
    success_status = true
  end
  return msg,success_status
end

def self.unsubscribe_me(channel, user)
  msg = []
  success_status = false
  return "$You can not unsubscribe your own #{channel.channel_or_cug}" if  channel.user_id == user.id

  subscription = Subscription.where(:channel_id => channel.id, :user_id => user.id).first
  if !subscription.nil?
    subscription.destroy
  else
    msg << "$You are not subscribed to #{channel.name}"
  end
  msg << "You have been successfully unsubscribed from #{channel.channel_or_cug} #{channel.name}" unless !msg.blank?
  success_status = true
  return msg,success_status
end

def self.unsubscribe_users(channel, users)
  msg = []
  success_status = false
  users.each do |email|
    email = User.valid_email(email)
    user = User.find_by_email email
    if !user.nil?
      if channel.user_id != user.id
        subscription = Subscription.where(:channel_id => channel.id, :user_id => user.id).first
        if !subscription.nil?
          subscription.destroy
          msg << "#{user.full_name} has been unsubscribed from #{channel.channel_or_cug} #{channel.name}"
          success_status = true
        else
          msg << "$#{user.full_name} has not subscribed to #{channel.channel_or_cug} #{channel.name}"
        end
      else
        msg << "$You can not unsubscribe your own #{channel.channel_or_cug}"
      end
    else
      msg << "$User with email id #{email} does not exist"
    end
  end
  if msg.blank?
    msg << "You have unsubscribed the users successfully"
    success_status = true
  end
  return msg, success_status
end
