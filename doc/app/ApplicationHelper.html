def autoresponse_msg(msg)
  message = ""
  return message if (msg.nil? || msg.blank?)
  if msg.is_a?(Array)
    msg.each do |m|
      message += "#{m[0,1]=='$' ? m[1,m.length] : m}" + ", "
    end
    message = message[0, message.length - 2]
  else
    message = "#{msg[0,1]=='$' ? msg[1,msg.length] : msg}"
  end
  message
end

def show_devise_msg(msg)
  message = ""
  return message if (msg.nil? || msg.blank?)

  if msg.is_a?(Array)
    msg.each do |m|
      message += "<span class='#{m[0,1]=='$' ? 'err_msg' : ''}'> #{m[0,1]=='$' ? m[1,m.length] : m}</span>" + ", "
    end
    message = message[0, message.length - 2]
  else
    message = "<span class='#{msg[0,1]=='$' ? 'err_msg' : ''}'>#{msg[0,1]=='$' ? msg[1,msg.length] : msg}</span>"
  end
  message.html_safe
end

def show_msg(msg)
  message = ""
  return message if (msg.nil? || msg.blank?)
  if msg.is_a?(Array) && msg.any?
    msg.each do |m|
      message += "<span class='#{m[0,1]=='$' ? 'err_msg' : ''}'> #{m[0,1]=='$' ? m[1,m.length] : m}</span>" + ", "
    end
    message = message[0, message.length - 2]
  else
    message = "<span class='#{msg[0,1]=='$' ? 'err_msg' : ''}'>#{msg[0,1]=='$' ? msg[1,msg.length] : msg}</span>"
  end
  message.html_safe
end
