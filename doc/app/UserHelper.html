def user_status_link(user)
  if user.is_active
    link_to image_tag("/assets/social/delete.png", :alt =>"Delete", :width =>"14", :height => "14", :border => "0"), deactivate_user_path(user)
  else
    link_to image_tag("/assets/social/accept.png", :alt =>"Delete", :width =>"14", :height => "14", :border => "0"), activate_user_path(user)
  end
end
